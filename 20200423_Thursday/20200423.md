```Ruby
# 686. Repeated String Match
# https://leetcode.com/problems/repeated-string-match/
# @param {String} a
# @param {String} b
# @return {Integer}
def repeated_string_match(a, b)
    repeat = 1
    while a.size <= b.size*2
        return repeat if a[b]
        a += a
        repeat += 1
    end
    -1
end
```

# @param {String[]} board

def valid_tic_tac_toe(board)
    @b = board.map(&:chars).flatten
    x_marks, y_marks, all_marks = 0,0,0
    x_turn = true
    stop = false
    return check(x_turn,stop,x_marks, y_marks, all_marks)
end

def check_win(marks)
    [7, 56, 448, 292, 146, 73, 273, 84].any?{|x| 0 != marks & x}
end

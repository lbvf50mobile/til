def houseRobber(nums)
    @dp = Array.new(nums.size)
    @n = nums
    [req(0), req(1)].max
end

def req(i)
    return 0 if i >= @n.size
    return @dp[i] if @dp[i]
    @dp[i] = @n[i] + [req(i+2),req(i+3)].max
end

# @param {Integer[]} tasks
# @return {Integer}
def minimum_rounds(tasks)
  h = {}
  tasks.each do |t|
    h[t] ||= 0
    h[t] += 1
  end
  counter = 0
  h.keys.each do |i|
    x = h[i]
    return -1 if 1 == x
    counter += x/3
    counter += 1 if 0 != x%3
  end
  return counter
end

def erase_overlap_intervals(intervals)
  # Based on:
  # https://leetcode.com/problems/non-overlapping-intervals/solution/
  intervals.sort_by!{|x| x[1]}
  ans = 0
  k = -Float::INFINITY
  intervals.each do |x,y|
    if x >= k
      k = y
    else
      ans += 1
    end
  end
  return ans
end

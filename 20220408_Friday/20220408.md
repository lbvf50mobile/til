class KthLargest

=begin
    :type k: Integer
    :type nums: Integer[]
=end
    def initialize(k, nums)
      @heap = HeapLike.new(k,nums)
    end


=begin
    :type val: Integer
    :rtype: Integer
=end
    def add(val)
      @heap.push_and_get(val)
    end
end

class HeapLike
  def initialize(k, nums)
    @d = false
    @k = k
    @n = nums.sort
    if @n.size >  @k
      @n = @n[-k..-1]
    end
    puts "Start: #{@n.inspect}" if @d
  end
  def push_and_get(x)
    @n.push(x)
    @n.sort!
    @n.shift if @n.size > @k
    @n.first
  end
end

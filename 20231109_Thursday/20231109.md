  ans = 0
  streak = 0
  mod = (10**9) + 7

    end
    ans = (ans + streak) % mod
  end
  return ans

package main

func countHomogenous(s string) int {
	ans, stk, mod := 0, 0, 1000_000_007
	for i, _ := range s {
		if 0 == i || s[i] == s[i-1] {
			stk += 1
		} else {
			stk = 1
		}
		ans = (ans + stk) % mod
	}
	return ans
}

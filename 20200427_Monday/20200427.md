```Python
def sumin(n):
    column = lambda i : (1+i)*i//2 + (i*(n-i))
    return sum( column(i) for i in range(1,n+1))
    
def sumax(n):
    column = lambda i : (i+n)*(n-(i-1))//2 + (i*(i-1))
    return sum(column(i) for i in range(1,n+1))
    
def sumsum(n):
    return sumin(n) + sumax(n)
```

    create_hist = ->x{ x.chars.each_with_object(Hash.new(0)){|x,obj| obj[x] ||= 0; obj[x] += 1}}
    hist = create_hist[text]
    balloon = create_hist['balloon']

    counter = hist[?b]/balloon[?b]

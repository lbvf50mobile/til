def gdx(a,b) # get dx;
  b[0] - a[0]
end
def gdy(a,b) # get dy;
  b[1] - a[1]
end

def check_straight_line(coordinates)
  # Based on:
  # https://leetcode.com/problems/check-if-it-is-a-straight-line/solution/
  c = coordinates
  dx = gdx(c[0], c[1])
  dy = gdy(c[0], c[1])
  (2...c.size).each do |i|
    if dx * gdy(c[0],c[i]) != dy * gdx(c[0],c[i])
      return false
    end
  end
  return true
end

package main

func gdx(a, b []int) int {
	return b[0] - a[0]
}
func gdy(a, b []int) int {
	return b[1] - a[1]
}
func checkStraightLine(coordinates [][]int) bool {
	c := coordinates
	dx := gdx(c[0], c[1])
	dy := gdy(c[0], c[1])
	for i := 2; i < len(c); i += 1 {
		one := dx * gdy(c[0], c[i])
		two := dy * gdx(c[0], c[i])
		if one != two {
			return false
		}
	}
	return true

}

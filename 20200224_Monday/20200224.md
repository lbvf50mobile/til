# 492. Construct the Rectangle
# https://leetcode.com/problems/construct-the-rectangle
# @param {Integer} area
# @return {Integer[]}
def construct_rectangle(area)
    require 'prime'
    if Prime.prime?(area)
        p "boom #{area}"
        return [area,1]
    end
    x = Math.sqrt(area)
    if 0 == x%1
        return [x.to_i,x.to_i]
    else
        while 1
            x = x.to_i
            y = area/x
            return [y,x] if area == x*y
            x -= 1
        end
    end
end

def diagonal_sort(mat)
  # Code based on:
  # https://leetcode.com/problems/sort-the-matrix-diagonally/discuss/489749/JavaPython-Straight-Forward
  m, n = mat.size, mat[0].size
  d = {}
  (0...m).each do |i|
    (0...n).each do |j|
      d[i-j] ||= []
      d[i-j].push(mat[i][j])
    end
  end
  d.keys.each do |key|
    d[key] = d[key].sort_by{|x| -x}
  end
  (0...m).each do |i|
    (0...n).each do |j|
      mat[i][j] = d[i-j].pop
    end
  end
  mat
end

# @param {Integer[][]} mat
# @return {Integer}
def diagonal_sum(mat)
  n = mat.size
  ans = 0
  (0...n).each do |i|
    ans += mat[i][i]
    ans += mat[i][n-1-i]
  end
  if n.odd?
    ans -= mat[n/2][n/2]
  end
  return ans
end

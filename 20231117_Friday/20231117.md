# @param {Integer[]} nums
# @return {Integer}
def min_pair_sum(nums)
  # Maxiest with smallest => minimized maximum.
  nums.sort!
  i,j = 0, nums.size - 1
  max = nums[i] + nums[j]
  i += 1
  j -= 1
  while i < j
    tmp = nums[i] + nums[j]
    max = tmp if tmp > max
    i += 1
    j -= 1
  end
  return max
end

def maximum_element_after_decrementing_and_rearranging(arr)
  # Based on:
  # https://leetcode.com/problems/maximum-element-after-decreasing-and-rearranging/solution/
  arr.sort!
  ans = 1
  (1...arr.size).each do |i|
    ans += 1 if arr[i] >= ans + 1
  end
  return ans
end

# class ListNode
#     attr_accessor :val, :next
#     def initialize(val)
#         @val = val
#         @next = nil
#     end
# end

# @param {ListNode} head
# @return {Boolean}
def hasCycle(head)
  return false if head.nil?
  h = {}
  h[head] = true
  while head
    head = head.next
    return true if head && h[head]
    h[head] = true if head
  end
  return false
end

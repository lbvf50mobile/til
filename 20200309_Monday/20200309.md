# Definition for a binary tree node.
# class TreeNode
#     attr_accessor :val, :left, :right
#     def initialize(val)
#         @val = val
#         @left, @right = nil, nil
#     end
# end

# @param {TreeNode} root
# @return {Integer}
def get_minimum_difference(root)
    queue = [root]
    min = nil
    
    while ! queue.empty?
        x = queue.shift
        min = x.val if x.val < min
        max = x.val if x.val > max
        queue.push(x.left) if x.left
        queue.push(x.right) if x.right
    end
    min
end

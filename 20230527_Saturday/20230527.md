def stone_game_iii(stone_value)
  # Based on:
  # https://leetcode.com/problems/stone-game-iii/solution/
  n = stone_value.size
  sv = stone_value
  dp = Array.new(n+1,0)
  (0...n).reverse_each do |i|
    dp[i] = sv[i] - dp[i+1]
    if i + 2 <= n
      x = sv[i] + sv[i+1] - dp[i+2]
      dp[i] = x if dp[i] < x
    end
    if i + 3 <= n
      x = sv[i] + sv[i+1] + sv[i+2] - dp[i+3]
      dp[i] = x if dp[i] < x
    end
  end
  return "Tie" if 0 == dp[0]
  return "Alice" if 0 < dp[0]
  return "Bob" if 0 > dp[0]
  raise "Unknown"
end

func stoneGameIII(stoneValue []int) string {
	n := len(stoneValue)
	v := stoneValue
	dp := make([]int, n+1)
	x := 0
	for i := n - 1; i >= 0; i -= 1 {
		dp[i] = v[i] - dp[i+1]
		x = dp[i]
		if i+2 <= n {
			x = v[i] + v[i+1] - dp[i+2]
            // condition must be here.
            // First dp update.
			if x > dp[i] {
				dp[i] = x
			} // Was  an error beause false optimisation.
            // During the flor loop dp updated twice.
		}
		if i+3 <= n {
			x = v[i] + v[i+1] + v[i+2] - dp[i+3]
            // condition must be here.
            // Second dp update.
			if x > dp[i] {
				dp[i] = x
			}
		}
	}
	if 0 < dp[0] {
		return "Alice"
	} else if 0 > dp[0] {
		return "Bob"
	} else {
		return "Tie"
	}
}

# 480. Sliding Window Median
# https://leetcode.com/problems/sliding-window-median/
# @param {Integer[]} nums
# @param {Integer} k
# @return {Float[]}

def median_sliding_window(nums, k)
    return [] if nums.empty? || k.zero?
    nums.map!(&:to_f)
    return nums if 1 == k
    odd = k.odd?
    queue = nums[0...k]
    answer = []
    median = ->(){
        x = queue.sort
        if odd
            return x[k/2]
        else
            return (x[k/2] + x[k/2-1])/2
        end
    }
    answer.push(median.())
    (k...nums.size).each do |i|
        queue.shift
        queue.push(nums[i])
        answer.push(median.())
    end
    answer 
end

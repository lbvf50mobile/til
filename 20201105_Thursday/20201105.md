# Leetcode: 1560. Most Visited Sector in a Circular Track.

Ruby code:
```Ruby
# Leetcode: 1560. Most Visited Sector in a Circular Track.
# https://leetcode.com/problems/most-visited-sector-in-a-circular-track/
# @param {Integer} n
# @param {Integer[]} rounds
# @return {Integer[]}
def most_visited(n, rounds)
    visited = [0] * (n+1) # storing number of visits at sector with number equal to index.
    (1...rounds.size).each do |i|
        current = rounds[i-1]
        finish = rounds[i]
        p "Start: #{current} finish: #{finish}"
        while true do
           visited[current] += 1
            p current
           break if current == finish
            current += 1
            current = 1 if current > n
        end
    end
    x = (1..n).to_a.sort_by{|x| visited[x]}
end
```

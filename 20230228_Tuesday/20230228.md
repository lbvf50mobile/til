# @param {TreeNode} root
# @return {TreeNode[]}
def find_duplicate_subtrees(root)
  # Based on:
  # https://leetcode.com/problems/find-duplicate-subtrees/solution/
  @cnt = {}
  @ans = []
  trv(root)
  return @ans
end

def trv(x)
  return "" if  !x
  rep = "(#{trv(x.left)})#{x.val}(#{trv(x.right)})"
  @cnt[rep] ||= 0
  @cnt[rep] += 1
  if 2 == @cnt[rep]
    @ans.push(x)
  end
  return rep
end

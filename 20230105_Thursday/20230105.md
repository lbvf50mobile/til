import "sort"
func findMinArrowShots(points [][]int) int {
  counter, last_arrow := 0, 0
  sort.Slice(points,func(i, j int) bool{
    return points[i][1] < points[j][1]
   })
   for _,v := range points {
     l,r := v[0],v[1]
     if 0 == counter || last_arrow < l {
       counter += 1
       last_arrow = r
     }
   }
  return counter
}

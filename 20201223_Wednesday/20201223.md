# 1664. Ways to Make a Fair Array.
# https://leetcode.com/problems/ways-to-make-a-fair-array/
# @param {Integer[]} nums
# @return {Integer}
def ways_to_make_fair(nums)
    tmp = [0] + nums
    even, odd = 0,0
    tmp1 = [[0,0]]
    (1...tmp.size).each do |i|
        if i.even?
            even += tmp[i]
        else
            odd += tmp[i]
        end
        tmp1.push([even,odd])
    end
    fair_count = 0
    (1...tmp1.size).each do |i|
        even_sum = tmp1[i-1][0] + odd - tmp1[i][1]
        odd_sum = tmp[i-1][1] + even - tmp1[i][0]
        fair_count += 1 if even_sum == odd_sum
    end
    fair_count 
end

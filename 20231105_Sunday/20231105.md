# @param {Integer[]} arr
# @param {Integer} k
# @return {Integer}
def get_winner(arr, k)
  # Based on:
  # https://leetcode.com/problems/find-the-winner-of-an-array-game/solution/
  max_element = arr.max
  queue = arr[1..-1] || []
  curr = arr[0]
  winstreak = 0
  while ! queue.empty?
    opponent = queue.shift
    if curr > opponent
      queue.push(opponent)
      winstreak += 1
    else
      queue.push(curr)
      curr = opponent
      winstreak = 1
    end
    if k == winstreak || curr == max_element
      return curr
    end
  end
  raise "Cannot be here."
end

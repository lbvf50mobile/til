  max_element = arr.max
  queue = arr[1..-1] || []
  curr = arr[0]
  winstreak = 0
  while ! queue.empty?
    opponent = queue.shift
    if curr > opponent
      queue.push(opponent)
      winstreak += 1
    else
      queue.push(curr)
      curr = opponent
      winstreak = 1

    end

package main

func getWinner(arr []int, k int) int {
	// Based on:
	// https://leetcode.com/problems/find-the-winner-of-an-array-game/solution/
	max := getMax(arr)
	curr := arr[0]
	winstreak := 0
	queue := make([]int, len(arr)-1)
	copy(queue, arr[1:])
	for 0 < len(queue) {
		opponent := queue[0]
		queue = queue[1:]
		if curr > opponent {
			queue = append(queue, opponent)
			winstreak += 1
		} else {
			queue = append(queue, curr)
			curr = opponent
			winstreak = 1
		}
		if winstreak == k || curr == max {
			return curr
		}
	}
	return -1
}

func getMax(x []int) int {
	ans := x[0]
	for _, v := range x {
		if v > ans {
			ans = v
		}
	}
	return ans
}

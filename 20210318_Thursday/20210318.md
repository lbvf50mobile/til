# Leetcode: 1574. Shortest Subarray to be Removed to Make Array Sorted.
# https://leetcode.com/problems/shortest-subarray-to-be-removed-to-make-array-sorted/
# @param {Integer[]} arr
# @return {Integer}
# fails: [1,2,3,10,0,7,8,9]
def find_length_of_shortest_subarray(arr)
    return 0 if 1 == arr.size
    i = 0
    j = arr.size - 1
    while i < arr.size - 1 && arr[i] <= arr[i+1]; i+=1; end
    return 0 if i == arr.size - 1
    while j >= 1 && arr[j-1] <= arr[j]; j -= 1; end
    # Ok hence i and j points at the ends of the sorted areas.
   
    # Now move i left till find the appropriate condidtions.
    move_i = i
    while move_i >= 0 && arr[move_i] >  arr[j]; move_i -= 1; end
    left = arr.size - (move_i + 1) - (arr.size - j)
    
    # Nove move j right till find the approprite condition.
    move_j = j
    while move_j < arr.size && arr[i] > arr[move_j]; move_j += 1; end
    right = arr.size - (i+1) - (arr.size - move_j)
    
    [left,right].min
end

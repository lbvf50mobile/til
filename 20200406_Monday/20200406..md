# 173. Binary Search Tree Iterator
# https://leetcode.com/problems/binary-search-tree-iterator/
# Definition for a binary tree node.
# class TreeNode
#     attr_accessor :val, :left, :right
#     def initialize(val)
#         @val = val
#         @left, @right = nil, nil
#     end
# end

class BSTIterator

=begin
    :type root: TreeNode
=end
    def initialize(root)
        @root_nil if root.nil
        return if @root_nil
        
        # find the right most
        @rt = root
        while ! @rt.right.nil?
            @rt = @rt.right
        end
        @stack = []
        element = {node: root, parent: nil, action: "right"}
        @stack.push(element)
        
    end


=begin
    @return the next smallest number
    :rtype: Integer
=end
    def next()
        return false if @root_nil
        
    end


=begin
    @return whether we have a next smallest number
    :rtype: Boolean
=end
    def has_next()
        return false if @root_nil
        return true
    end


end

# param_2 = obj.has_next()

def sum_numbers(root)
  return dfs(root).sum{|x| x.to_i}
end

def dfs(x)
  return [] if x.nil?
  return [x.val.to_s] if (!x.left) && (!x.right)
  tmp = dfs(x.left) + dfs(x.right)
  return tmp.map{|n| x.val.to_s + n}
end

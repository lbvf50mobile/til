# class ListNode
#     attr_accessor :val, :next
#     def initialize(val)
#         @val = val
#         @next = nil
#     end
# end

# @param {ListNode} head
# @return {ListNode}
def detectCycle(head)
  map = {}
  counter = 0
  while head
    if ! map[head.object_id]
      counter += 1
      map[head.object_id] = counter
    else
      return head
    end
    head = head.next
  end
  return -1
end

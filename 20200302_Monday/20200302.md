# 504. Base 7
# https://leetcode.com/problems/base-7/
# @param {Integer} num
# @return {String}
def convert_to_base7(num)
    return"0" if num.zero?
    negative = num < 0 
    num = - num if negative
    answer = ""
    while num > 0
        x = num % 7
        answer = x.to_s + answer
        num /= 7
    end
    answer = "-" + answer if negative
    answer
end

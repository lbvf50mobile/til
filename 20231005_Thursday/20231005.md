package main

func majorityElement(nums []int) []int {
	h := make(map[int]int)
	bound := len(nums) / 3
	for _, k := range nums {
		h[k] += 1
	}
	ans := make([]int, 0, 2)
	for k, v := range h {
		if v > bound {
			ans = append(ans, k)
		}
	}
	return ans
}

# @return {String[]}
def build_array(target, n)
  # Based on the Leetcode's solution:
  # https://leetcode.com/problems/build-an-array-with-stack-operations/solution/
  ans = []
  i = 0
  target.each do |num|
    while i < num - 1
      ans.push("Push")
      ans.push("Pop")
      i += 1
    end
    ans.push("Push")
    i += 1
  end
  return ans
end

# @return {String}
def longest_palindrome(s)
  # Based on:
  # https://leetcode.com/problems/longest-palindromic-substring/solution/
  n = s.size
  dp = Array.new(n){ Array.new(n, false)}
  ans = [0,0]
  n.times do |i|
    dp[i][i] = true
  end
  (n-1).times do |i|
    if s[i] == s[i+1]
      dp[i][i+1] = true
      ans = [i, i + 1]
    end
  end
  (2...n).each do |diff|
    (0...n-diff).each do |i|
      j = i + diff
      if s[i] == s[j] && dp[i+1][j-1]
        dp[i][j] = true
        ans = [i,j]
      end
    end
  end
  i,j = ans
  return s[i..j]
end

package main

func longestPalindrome(s string) string {
	// Based on:
	// https://leetcode.com/problems/longest-palindromic-substring/solution/
	n := len(s)
	dp := make([][]bool, n)
	ans := []int{0, 0}
	for i, _ := range dp {
		dp[i] = make([]bool, n)
	}
	for i := 0; i < n; i += 1 {
		dp[i][i] = true
	}
	for i := 0; i < n-1; i += 1 {
		if s[i] == s[i+1] {
			dp[i][i+1] = true
			ans[0], ans[1] = i, i+1
		}
	}
	for diff := 2; diff < n; diff += 1 {
		for i := 0; i < n-diff; i += 1 {
			j := i + diff
			if s[i] == s[j] && dp[i+1][j-1] {
				dp[i][j] = true
				ans[0], ans[1] = i, j
			}
		}
	}
	return s[ans[0] : ans[1]+1]
}

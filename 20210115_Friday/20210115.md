def solve(st):
    x = list(st)
    for i in range(0,len(x)//2):
        # print(x[i],x[len(x)- i - 1])
        a = ord(x[i])
        b = ord(x[len(x)- i - 1])
        print(a,b,abs(a-b)
        if not ( abs(a-b) <= 2):

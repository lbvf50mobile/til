# Leetcode: 1405. Longest Happy String.
# https://leetcode.com/problems/longest-happy-string/
# @param {Integer} a
# @param {Integer} b
# @param {Integer} c
# @return {String}
# fails: 0, 8, 11
def longest_diverse_string(a, b, c)
    @x = [[?a,a], [?b, b], [?c,c]]
    ans = ""
    while true
        size = ans.size
        if ans.empty?
            ans += get_max("")
        else
            ans += get_max(ans[size-1])
        end
        break if  ans.size == size
    end
    ans
end

def get_max(not_symbol)
    max = 0
    char = ""
    index = -1
    (0...3).each do |i|
        sym,num = @x[i]
        if max < num && sym != not_symbol
            max = num
            char = sym
            index = i
        end
    end
    return "" if 0 == max
    if max >= 2
        @x[index][1] -= 2
        return char*2
    end
    @x[index][1] -= 1
    return char
end

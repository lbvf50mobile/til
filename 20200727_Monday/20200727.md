def mutateTheArray(n, a)
    prv = 0
    (0...a.size).each do |i|
        tmp = a[i]
        nxt = i < (a.size-1) ? a[i+1] : 0
        a[i] = a[i] + nxt + prv
        prv = tmp
    end

def hashMap(queryType, query)
    diff,sum = 0, 0
    total = 0
    hsh = {}
    action = {
        'insert' => ->x{ i,v = x; hsh[i-diff] = v-sum},
        'get' => ->x{ v = hsh[x[0]-diff] + sum ; total += v },
        'addToKey' => ->x{ diff += x[0]},
        'addToValue' => ->x{ sum += x[0]},
    }
    queryType.each_with_index do |q,i|
        action[q][query[i]]
    end
    total
end

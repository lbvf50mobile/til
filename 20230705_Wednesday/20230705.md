package main

import (
	"fmt"
)

func longestSubarray(nums []int) int {
	i, j := 0, 0
	n := len(nums)
	zeros := 0
	if 0 == nums[0] {
		zeros = 1
	}
	ans := 0
	for i < n {
		// Calculate.
		// Always take one element from lenght.
		l := j - i
		if zeros < 2 && ans < l {
			ans = l
		}
		// Condition to grow.
		// If there is a place to grow.
		// And less then two zeros.
		if zeros < 2 && j < n-1 {
			j += 1
			if 0 == nums[j] {
				zeros += 1
			}
			continue
		}
		// Condition to shirnk.
		// Two or more zeros.
		// No place to grow for j.
		if zeros >= 2 || n-1 == j {
			if 0 == nums[i] {
				zeros -= 1
			}
			i += 1
			continue
		}
	}
	return ans
}

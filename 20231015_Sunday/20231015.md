def num_ways(steps, arr_len)
  # Based on:
  # https://leetcode.com/problems/number-of-ways-to-stay-in-the-same-place-after-some-steps/solution/
  mod = 10**9 + 7
  len = [steps, arr_len].min
  dp = Array.new(len){ Array.new(steps+1,0)}
  dp[0][0] = 1
  (1..steps).each do |r|
    (0...len).reverse_each do |c|
      ans = dp[c][r-1]
      if c > 0
        ans = (ans + dp[c-1][r-1]) % mod
      end
      if c < len - 1
        ans = (ans + dp[c+1][r-1]) % mod
      end
      dp[c][r] = ans
    end
  end
  dp[0][0] = 1
  return dp[0][steps]
end

package main

func numWays(steps int, arrLen int) int {
	// Based on:
	// https://leetcode.com/problems/number-of-ways-to-stay-in-the-same-place-after-some-steps/solution/
	mod := 1000000007
	if steps < arrLen {
		arrLen = steps
	}
	dp := make([][]int, arrLen)
	for i := 0; i < arrLen; i += 1 {
		dp[i] = make([]int, steps+1)
	}
	dp[0][0] = 1
	for r := 1; r <= steps; r += 1 {
		for c := arrLen - 1; c >= 0; c -= 1 {
			ans := dp[c][r-1]
			if c > 0 {
				ans = (ans + dp[c-1][r-1]) % mod
			}
			if c < arrLen-1 {
				ans = (ans + dp[c+1][r-1]) % mod
			}
			dp[c][r] = ans
		}
	}
	dp[0][0] = 1
	return dp[0][steps]

}

# Vowels in a String.
# @param {String} s
# @return {String}
def sort_vowels(s)
  initial_array = s.chars
  indicies = []
  values = []
  initial_array.each_with_index do |char, i|
    if /[aeiouAEIOU]/ === char
      indicies.push(i)
      values.push(char)
    end
  end
  values.sort!
  indicies.size.times do |i|
    arr_i = indicies[i]
    val = values[i]
    initial_array[arr_i] = val
  end
  return initial_array.join
end

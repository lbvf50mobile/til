# 633. Sum of Square Numbers

# @param {Integer} c
# @return {Boolean}
def judge_square_sum(c)
    return true if 0 == c
    (1..1000_000).each do |a|
        x = c - a*a
        # p "#x = {x}, a = #{a}, c = #{c}"
        return false if x < 0
        return true if 0 == x
        return true if  0 == Math.sqrt(x)%1
    end
    raise "How it could happens?"
end

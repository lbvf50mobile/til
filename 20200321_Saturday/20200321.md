# Leetcode: 575. Distribute Candies.


```Ruby
# 575. Distribute Candies
# https://leetcode.com/problems/distribute-candies/
# @param {Integer[]} candies
# @return {Integer}
def distribute_candies(candies)
    answer = 0
    i = 0
    candies.sort!
    while i < candies.size
        if candies[i] == candies[i+1]
            i += 2
            answer += 1
        else
            i += 1
        end
    end
    answer
end
```

# 443. String Compression
# https://leetcode.com/problems/string-compression/
# @param {Character[]} chars
# @return {Integer}
def compress(chars)
    x = chars
        .chunk_while{|x,y| x == y}
        .map{|x| x.size == 1 ? [x[0]] : (x[0]+x.size.to_s).chars}
        .flatten
    x.each_with_index{|v,i| chars[i] = v}
    x.size
end

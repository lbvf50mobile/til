# @return {Integer[]}
def find_smallest_set_of_vertices(n, edges)
  # Based on:
  # https://leetcode.com/problems/minimum-number-of-vertices-to-reach-all-nodes/solution/
  out = Array.new(n,false)
  edges.each do |a,b|
    out[b] = true
  end
  ans = []
  (0...n).each do |i|
    ans.push(i) if ! out[i]
  end
  return ans
end

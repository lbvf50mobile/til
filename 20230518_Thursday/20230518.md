  out = Array.new(n,false)

  end
  return ans
end

package main

func findSmallestSetOfVertices(n int, edges [][]int) []int {
	out := make([]bool, n)
	for _, edge := range edges {
		_, b := edge[0], edge[1]
		out[b] = true
	}
	ans := []int{}
	for i := 0; i < n; i += 1 {
		if !out[i] {
			ans = append(ans, i)
		}
	}
	return ans
}

# class ListNode
#     attr_accessor :val, :next
#     def initialize(val)
#         @val = val
#         @next = nil
#     end
# end

# @param {ListNode} head
# @return {Boolean}
def hasCycle(head)
  h = {}
  tmp = head
  while tmp
    return true if h[tmp]
    h[tmp] = true
    tmp = tmp.next
  end
  return false
end

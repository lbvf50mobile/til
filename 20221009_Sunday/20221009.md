def find_target(root, k)
  @ans = false
  @k, @h = k, {}
  dfs(root)
  return @ans
end

def dfs(node)
  return true if @ans
  return false if node.nil?
  if @h[@k - node.val]
    @ans = true
    return
  end
  @h[node.val] = true
  dfs(node.left)
  dfs(node.right)
end

package main

func numberOfMatches(n int) int {
	ans := 0
	for n > 1 {
		if 0 == n%2 {
			ans += n / 2
			n = n / 2
		} else {
			ans += (n - 1) / 2
			n = (n-1)/2 + 1
		}
	}
	return ans
}

# @param {Integer[]} nums
# @param {Integer} k
# @return {Integer[]}
def top_k_frequent(nums, k)
  # Straight forward without Heap.
  h = nums.tally
  h.to_a.sort_by{|x| -x[1]}.map(&:first)[0...k]    
end

func numSubarraysWithSum(nums []int, goal int) int {
	// Take look prev_sum
	// Hint here.
	// https://leetcode.com/problems/binary-subarrays-with-sum/solution/
	h := make(map[int]int)
	ans := 0
	currSum := 0
	for _, v := range nums {
		currSum += v
		prevSum := currSum - goal // Here!
		if goal == currSum {
			ans += 1
		}
		ans += h[prevSum]
		h[currSum] += 1
	}
	return ans
}

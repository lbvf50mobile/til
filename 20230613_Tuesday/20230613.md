# @param {Integer[][]} grid
# @return {Integer}
def equal_pairs(grid)
  g = grid
  rows = g.map{|x| x.join(?,)}.tally # Was a bug here, a delimiter needed.
  cols = g.transpose.map{|x| x.join(?,)}.tally
  ans = 0 
  rows.keys.each do |hash|
    cols[hash] ||= 0
    ans += rows[hash] * cols[hash]
  end
  ans
end

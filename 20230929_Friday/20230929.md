end

def up(x)
  (1...x.size).each do |i|
    return false if x[i-1] > x[i]
  end

end

def down(x)
  (1...x.size).each do |i|
    return false if x[i-1] < x[i]
  end
  return true

# Leetcode: 896. Monotonic Array.

# Monotonic Array.
# 2023.09.29 Daily Challenge.
# @param {Integer[]} nums
# @return {Boolean}
def is_monotonic(nums)
  x = nums
  up = false
  down = false
  (1...x.size).each do |i|
    return false if up && down
    up = true if x[i-1] < x[i]
    down = true if x[i-1] > x[i]
  end
  return false if up && down
  return true
end

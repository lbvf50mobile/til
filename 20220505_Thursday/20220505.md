class MyStack
    def initialize()
      @q1, @q2 = [],[]
        
    end


=begin
    :type x: Integer
    :rtype: Void
=end
    def push(x)
      @q1.push x
    end


=begin
    :rtype: Integer
=end
    def pop()
       last = nil
       @q1.size.times do
         last = @q1.shift
         @q2.push last
       end
       (@q2.size-1).times do 
         @q1.push(@q2.shift)
       end
       @q2.shift
       last
    end


=begin
    :rtype: Integer
=end
    def top()
      x = pop()
      push(x)
      x
    end


=begin
    :rtype: Boolean
=end
    def empty()
      @q1.empty?
    end


end

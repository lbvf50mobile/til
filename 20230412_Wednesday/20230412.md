# @param {String} path
# @return {String}
def simplify_path(path)
  path = path.split(?/).select{|x| !x.empty?}.select{|x| x != '.'}
  ans = []
  path.each do |x|
    if '..' == x
      ans.pop
    else
      ans.push(x)
    end
  end
  return '/' if ans.empty?
  return '/' + ans.join(?/)
end

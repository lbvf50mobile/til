# 521. Longest Uncommon Subsequence I
# https://leetcode.com/problems/longest-uncommon-subsequence-i/
# @param {String} a
# @param {String} b
# @return {Integer}
def find_lu_slength(a, b)
    return -1 if a == b
    return [a.size,b.size].max
end

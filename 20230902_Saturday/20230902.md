# @param {String} s
# @param {String[]} dictionary
# @return {Integer}
def min_extra_char(s, dictionary)
  # Based on:
  # https://leetcode.com/problems/extra-characters-in-a-string/solution/
  d = dictionary.to_set
  n = s.size
  dp = Array.new(n+1,0)
  (0...n).reverse_each do |i|
    dp[i] = 1 + dp[i+1]
    (i...n).each do |j|
        curr = s[i..j]
        if d.include?(curr)
          dp[i] = [dp[i], dp[j+1]].min
        end
    end
  end
  return dp[0]
end

def continuous_subarrays(nums)
  # based on the Leetcode's solution.
  f = {}
  f.default = 0
  l, r = 0, 0
  c = 0
  while r < nums.size
    f[nums[r]] = f[nums[r]] + 1
    while (f.max)[0] - (f.min)[0] > 2 
      f[nums[l]] -= 1
      if 0 == f[nums[l]]
        f.delete(nums[l])
      end
      l += 1
    end
    c += r - l +1
    r +=1
  end
  return c
end

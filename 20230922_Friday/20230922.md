# @param {String} s
# @param {String} t
# @return {Boolean}
def is_subsequence(s, t)
  # S is Subsequence of T. S is less then T.
  return false if s.size > t.size
  i,j = 0, 0
  # Ok. Considiration time.
  # Need to reach the end of s skipping t elements.
  while i < s.size && j < t.size
    a,b = s[i], t[j]
    if a == b
      i += 1
      j += 1
    else
      j += 1
    end
  end
  return s.size == i 
end

#     def initialize(val = 0, left = nil, right = nil)

#         @right = right

# @param {TreeNode} root
# @return {Integer[]}
def find_mode(root)
  @c = {}
  dfs(root)
  return [] if @c.empty?
  max = @c.values.max
  @c.select{|i,x| x == max}.map(&:first)
end

def dfs(x)
  return if !x
  @c[x.val] ||= 0
  @c[x.val] += 1
  dfs(x.left)
  dfs(x.right)
end

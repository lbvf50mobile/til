def find_mode(root)
  @c = {}
  dfs(root)
  return [] if @c.empty?
  max = @c.values.max
  @c.select{|i,x| x == max}.map(&:first)
end

def dfs(x)
  return if !x
  @c[x.val] ||= 0
  @c[x.val] += 1
  dfs(x.left)
  dfs(x.right)
end

// Accepted.

// Memory Usage: 6.7 MB, less than 45.45% of Go online submissions for Find Mode

package main

/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
var c map[int]int

func findMode(root *TreeNode) []int {
	c = make(map[int]int)
	dfs(root)
	if 0 == len(c) {
		return []int{}
	}
	flag := true
	max := 0
	for _, v := range c {
		if flag {
			max = v
			flag = false
		}
		if v > max {
			max = v
		}
	}
	ans := make([]int, 0)
	for i, v := range c {
		if v == max {
			ans = append(ans, i)
		}
	}
	return ans
}

func dfs(root *TreeNode) {
	if nil == root {
		return
	}
	c[root.Val] += 1
	dfs(root.Left)
	dfs(root.Right)
}

# @param {String} s
# @return {String}
def to_goat_latin(s)
    vovels = /^[aeiou]/i
    arr = s.split(" ")
    arr.each_with_index.map{|x,i|
        if( vovels === x)
            x + "ma" + "a"*(i+1)
        else
            x[1..-1] +x[0] + 'ma' + ?a*(i+1)
        end
    }.join(" ")
end

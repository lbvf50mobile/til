# @param {String[]} words
# @param {Integer} k
# @return {String[]}
def top_k_frequent(words, k)
  hash = words.tally
  hash.to_a.sort{|x,y| x[1] == y[1] ? x[0]<=>y[0] : y[1] <=> x[1]}[0...k].map(&:first)
end

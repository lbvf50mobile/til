# @param {Integer[]} nums
# @return {String[]}
def summary_ranges(nums)
  return nums if nums.empty?
  # Array of pairs.
  ans = []
  ans.push([nums[0],nums[0]])
  (1...nums.size).each do |i|
    a = ans.last.last
    b = nums[i]
    if a + 1 == b
      ans.last[1] = b
    else
      ans.push([b,b])
    end
  end
  ans = ans.map{|a,b| 
    if a == b
      a.to_s
    else
      "#{a}->#{b}"
    end
  }
  return ans
end

# class TreeNode

#     def initialize(val = 0, left = nil, right = nil)

# @param {TreeNode} root
# @return {Integer}
def max_ancestor_diff(root)
  @diff = 0
  return 0 if ! root
  dfs(root, root.val, root.val)
  return @diff
end
def dfs(r,min,max)
  return if ! r
  min = r.val if r.val < min
  max = r.val if r.val > max
  tmp = (max - min).abs
  @diff = tmp if tmp > @diff
  dfs(r.left, min, max)
  dfs(r.right, min,max)
end

# Leetcode: 44. Wildcard Matching.
# https://leetcode.com/problems/wildcard-matching/
# @param {String} s
# @param {String} p
# @return {Boolean}
# Fail on ""
# "******"
def is_match(s, pt)
    @s, @pt = s.chars, pt.chars
    @dp = Array.new(s.size).map{ Array.new(pt.size)}
    rec(0,0)
end

def rec(i,j)
    if i == @s.size && j == @pt.size
        return true
    end
    if i == @s.size || j == @pt.size
        return false
    end
    return @dp[i][j] if ! @dp[i][j].nil?
    if ?? == @pt[j]
        @dp[i][j] = rec(i+1, j+1)
        return @dp[i][j]
    end
    if ?* == @pt[j]
        ans = false
        (i..@s.size).each do |ii|
            ans = rec(ii,j+1)
            break if ans
        end
        @dp[i][j] = ans
        return @dp[i][j]
        
    end
     if @s[i] == @pt[j]
        @dp[i][j] = rec(i+1,j+1)
        return @dp[i][j]
    end
    if @s[i] != @pt[j]
        @dp[i][j] = false
        return @dp[i][j]
    end
end

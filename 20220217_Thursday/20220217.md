def combination_sum(candidates, target)
  @d = false
  @c,@t = candidates, target
  @ans = []
  (0...@c.size).each do |i|
    back(i,0,[])
  end
  @ans
end


def back(i,sum,arr)
  # Add.
  # Check.
  # Move.
  return if i >= @c.size
  sum += @c[i]
  arr.push @c[i]
  if @t == sum
    puts "eq #{i}, #{sum}, #{arr.inspect}" if @d
    @ans.push(arr.clone)
    arr.pop
    return
  end
  if @t < sum
    arr.pop
    return
  end
  (i...@c.size).each do |j|
    back(j,sum,arr)
  end
  arr.pop
end

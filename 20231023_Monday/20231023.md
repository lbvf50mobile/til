def is_power_of_four(n)
  return false if n <= 0
  # This means in binary format n must be 1 and even number of zeros.
  while 0 == n%4
    n /= 4
  end
  n == 1
end

package main

func isPowerOfFour(n int) bool {
	if 1 == n {
		return true
	}
	if 0 >= n {
		return false
	}
	for 0 == n%4 {
		n /= 4
	}
	return 1 == n
}

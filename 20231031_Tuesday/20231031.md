# @param {Integer[]} pref
# @return {Integer[]}
def find_array(pref)
  # Got tip from the Leetcode's solution:
  # https://leetcode.com/problems/find-the-original-array-of-prefix-xor/solution/
  ans = Array.new(pref.size,pref[0])
  (1...pref.size).each do |i|
    ans[i] = pref[i] ^ pref[i-1]
  end
  return ans
end

#         @val = val

#         @right = right
#     end
# end
# @param {TreeNode} root
# @return {Integer}
def longest_zig_zag(root)
  # Based on @longluo's solution.
  # https://leetcode.com/problems/longest-zigzag-path-in-a-binary-tree/solution/1868016
  # From DFS to BFS.
  @max = 0
  q = [[root,0,0]]
  while !q.empty?
    x,l,r = q.shift
    next if !x
    @max = l if l > @max
    @max = r if r > @max
    q.push([x.left,r+1,0]) if x.left
    q.push([x.right,0,l+1]) if x.right
  end
  return @max
end

func dfs(x *TreeNode, lft bool, n int) {
	if nil == x {
		return
	}
	if max < n {
		max = n
	}
	if lft {
		dfs(x.Left, true, 1)
		dfs(x.Right, false, n+1)
	} else {
		dfs(x.Left, true, n+1)
		dfs(x.Right, false, 1)
	}
}

def array_sign(nums)
  product = nums.reduce(&:*)
  return -1 if 0 > product
  return 0 if 0 == product
  return 1 if 0 < product
end

package main

func arraySign(nums []int) int {
  pr := 1
  for _, v := range nums {
    x := 1
    if v < 0 {
      x = -1
    } else if 0 == v {
      x = 0
    } else {
      x = 1
    }
    pr *= x
  }
  return pr
}

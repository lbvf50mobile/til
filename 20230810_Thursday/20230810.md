# @param {Integer[]} nums
# @param {Integer} target
# @return {Boolean}
def search(nums, target)
  # Based on:
  # https://leetcode.com/problems/search-in-rotated-sorted-array-ii/discuss/1891315/Python-or-binary-search
  n,t = nums, target
  l, r = 0, n.size - 1
  while l <= r
    m = l + (r - l)/2
    return true if t == n[l]
    return true if t == n[r]
    return true if t == n[m]
    if n[m] == n[l] || n[m] == n[r]
      l += 1
      r -= 1
    elsif n[l] <= n[m]
      if n[l] < t && t < n[m]
        r = m - 1
      else
        l = m + 1
      end
      next
    else
      if n[m] < t && t < n[r]
        l = m + 1
      else
        r = m - 1
      end
    end
  end
  return false
end

# @param {Integer[]} nums
# @return {Integer[]}
def sort_array_by_parity(nums)
  # Two pointers.
  e = 0 # Even pointer.
  o = nums.size - 1 # Odd pointer.
  ans = Array.new(nums.size)
  nums.each do |v|
    break if e > o
    if v.even?
      ans[e] = v
      e += 1
    else
      ans[o] = v
      o -= 1
    end
  end
  return ans
end

#         @val = val
#         @next = _next
#     end
# end
# @param {ListNode} head
# @return {ListNode}
def odd_even_list(head)
  odd = []
  even = []
  tmp = head
  i = 1
  while tmp 
    if i.even?
      even.push(tmp)
    else
      odd.push(tmp)
    end
    tmp = tmp.next
    i += 1
  end
  ans = odd + even
  ans.each{|x| x.next = nil}
  (0...ans.size-1).each{|i| ans[i].next = ans[i+1]} 
  return ans[0]
end

 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func oddEvenList(head *ListNode) *ListNode {
  if nil == head { return head}
  odd := &ListNode{}
  even := &ListNode{}
  ot,et,tmp := odd, even, head
  for i := 1; nil != tmp ; i, tmp = i + 1, tmp.Next {
    if 1 == i%2 {
      ot.Next = tmp
      ot = tmp
    }else{
      et.Next = tmp
      et = tmp
    }
  }
  ot.Next, et.Next = nil, nil
  if nil != odd.Next {
    ot.Next = even.Next
  
  }
  return odd.Next
}

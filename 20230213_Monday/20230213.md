def count_odds(low, high)
  size = high - low + 1
  if low.even?
    # 10 - 8 + 1 = 3; 3-1 = 2; 2/2 = 1;
    size -= 1 if size.odd?
  else
    # 7-3+1 = 5; 5+1 = 6; 6/2 = 3;
    size += 1 if size.odd?
  end
  size/2
end

func countOdds(low int, high int) int {
  size := high - low + 1
  if 0 == low%2 {
    // 10 - 8 + 1 = 3; 3-1 = 2; 2/2 = 1;
    if 0 != size%2 {
      size -= 1
    }
  }else{
    // 7-3+1 = 5; 5+1 = 6; 6/2 = 3;
    if 0 !=  size%2{
      size += 1
    }
  }
  return size/2;
}

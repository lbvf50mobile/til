# @param {String[]} words
# @param {String} order
# @return {Boolean}
def is_alien_sorted(words, order)
  h = order.chars.zip((?a..?z).to_a).to_h
  convert = ->s{ s.chars.map{|x| h[x]}.join}
  nw = words.map(&convert)
  (1...nw.size).each do |j|
    i = j-1
    return false if nw[i] > nw[j]
  end
  return true
end

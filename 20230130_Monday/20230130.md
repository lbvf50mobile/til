# @param {Integer} n

def tribonacci(n)
  a, b, c = 0, 1, 1
  return a if 0 == n
  return b if 1 == n
  return c if 2 == n
  i = 3
  while i <= n 
    a,b,c = b, c, a+b+c
    i += 1
  end
  return c
end

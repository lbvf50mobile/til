# @param {Integer[]} nums
# @param {Integer} target
# @return {Integer}
def search(nums, target)
    return -1 if nums.empty?
    lo = 0
    hi = nums.size - 1
    while lo < hi
        mid = (lo+hi)/2
        return mid if target == nums[mid]
        if target < nums[mid]
            hi = mid-1
        else
            lo = mid + 1
        end
    end
    nums[hi] == target ? hi : -1
    
end

def xmastree(n):
    ans = []
    for i in range(1,n+1):
        ans.append(boom(i,n))
    ans.append(boom(1,n))
    ans.append(boom(1,n))
    return ans
def boom(n,h):
    return a(n,h) + b(n) + "#" + b(n)  + a(n,h)
def a(n,h):
    return ((h-n)*"_") 
def b(n):
    return (n-1) * "#"

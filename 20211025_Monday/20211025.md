Calculate min on every pop if value is min.

class MinStack
    def initialize()
      @a = []
      @min = nil
        
    end


=begin
    :type val: Integer
    :rtype: Void
=end
    def push(val)
      @min = val if ! @min
      @min = val if val < @min
      @a.push(val)
        
    end


=begin
    :rtype: Void
=end
    def pop()
      return if @a.empty?
      x = @a.pop()
      if x == @min
        @min = @a.min
      end
      return x
    end


=begin
    :rtype: Integer
=end
    def top()
      @a.last
    end


=begin
    :rtype: Integer
=end
    def get_min()
      @min
    end


end

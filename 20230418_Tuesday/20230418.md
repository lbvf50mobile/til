# @param {String} word1
# @param {String} word2
# @return {String}
def merge_alternately(word1, word2)
  flag = true
  i,j = 0,0
  ans = []
  while i < word1.size && j < word2.size
    if flag
      ans.push(word1[i])
      i += 1
    else
      ans.push(word2[j])
      j += 1
    end
    flag = !flag
  end
  while i < word1.size
    ans.push(word1[i])
    i += 1
  end
  while j < word2.size
    ans.push(word2[j])
    j += 1
  end
  ans.join
end

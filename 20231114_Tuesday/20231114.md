require "set"
def count_palindromic_subsequence(s)
  # Based on:
  # https://leetcode.com/problems/unique-length-3-palindromic-subsequences/solution/
  first = Array.new(26,-1)
  last = Array.new(26,-1)
  s.chars.each_with_index do |char,i|
    curr = char.ord - 'a'.ord
    first[curr] = i if -1 == first[curr]
    last[curr] = i
  end
  ans = 0
  26.times do |i|
    next if -1 == first[i]
    between = Set.new
    (first[i] + 1...last[i]).each do |j|
      between.add(s[j])
    end
    ans += between.size
  end
  return ans
end

class Solution:
    # Copied from:
    # https://leetcode.com/problems/unique-length-3-palindromic-subsequences/solution/
    def countPalindromicSubsequence(self, s: str) -> int:
        first = [-1] * 26
        last = [-1] * 26
        
        for i in range(len(s)):
            curr = ord(s[i]) - ord("a")
            if first[curr] == -1:
                first[curr] = i
            
            last[curr] = i
        
        ans = 0
        for i in range(26):
            if first[i] == -1:
                continue
                
            between = set()
            for j in range(first[i] + 1, last[i]):
                between.add(s[j])
            
            ans += len(between)

        return ans

def can_make_arithmetic_progression(arr)
  a = arr.sort
  d = a[1] - a[0]
  (2...a.size).each do |i|
    return false if a[i] - a[i-1] != d
  end
  return true
end

package main

import "sort"

func canMakeArithmeticProgression(arr []int) bool {
	a := arr
	sort.Ints(a)
	d := a[1] - a[0]
	for i := 2; i < len(a); i += 1 {
		if a[i]-a[i-1] != d {
			return false
		}
	}
	return true
}

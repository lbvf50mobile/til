# @param {Integer[]} arr
# @return {Boolean}
def can_make_arithmetic_progression(arr)
  a = arr.sort
  d = a[1] - a[0]
  (2...a.size).each do |i|
    return false if a[i] - a[i-1] != d
  end
  return true
end

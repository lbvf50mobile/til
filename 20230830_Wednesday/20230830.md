# @param {Integer[]} nums
# @return {Integer}
def minimum_replacement(nums)
  # Based on:
  # Leetcode's soltuion.
  answer = 0
  n = nums.size
  i = n - 2
  while i >= 0
    if nums[i] > nums[i+1]
      num_elements = (nums[i] + nums[i+1] - 1)/nums[i+1]
      answer += num_elements - 1
      nums[i] = nums[i]/num_elements
    end
    i -= 1
  end
  return answer
end

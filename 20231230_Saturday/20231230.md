package main

func makeEqual(words []string) bool {
	counter := make([]int, 26)
	n := len(words)
	// Fill the counter.
	for _, w := range words {
		for _, c := range w {
			counter[int(c-'a')] += 1
		}
	}
	for _, count := range counter {
		if 0 != count%n {
			return false
		}
	}
	return true
}

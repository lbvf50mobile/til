package main

func findSpecialInteger(arr []int) int {
	size := len(arr)
	quadro := float32(size) / 4.0
	counter := 0
	for i, v := range arr {
		if 0 == i || arr[i-1] != v {
			counter = 0
		}
		counter += 1
		if float32(counter) > quadro {
			return v
		}
	}
	panic("Could not reach this point.")
}

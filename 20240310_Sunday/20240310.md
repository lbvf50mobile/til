package main

func intersection(nums1 []int, nums2 []int) []int {
	h1, ansMap := make(map[int]bool), make(map[int]bool)
	for _, v := range nums1 {
		h1[v] = true
	}
	for _, v := range nums2 {
		if h1[v] {
			ansMap[v] = true
		}
	}
	ans := make([]int, 0, len(ansMap))
	for k, _ := range ansMap {
		ans = append(ans, k)
	}
	return ans
}

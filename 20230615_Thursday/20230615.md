#     end

# @param {TreeNode} root
# @return {Integer}
def max_level_sum(root)
  q = [[root,1]]
  max = root.val
  l = 1
  while !q.empty?
    sum = 0
    nq = []
    current_level = q[0].last 
    q.each do |node,lvl|
      sum += node.val 
      nq.push([node.left,lvl+1]) if node.left
      nq.push([node.right,lvl+1]) if node.right
    end
    q = nq
    if max < sum 
      max = sum
      l = current_level
    end
  end
  l
end

# @param {TreeNode} root

def max_level_sum(root)
  h = {}
  q = [[root,1]]
  while !q.empty?
    x,l = q.shift
    h[l] ||= 0
    h[l] += x.val
   q.push([x.left,l+1]) if x.left
   q.push([x.right,l+1]) if x.right
  end
  max = h.values.max
  h.to_a.select{|x| x.last == h.values.max}.sort_by(&:first).first.first 
end

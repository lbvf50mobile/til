package main

func canConstruct(s string, k int) bool {
	if k > len(s) {
		return false
	}
	freq := [26]int{}
	for _, v := range s {
		i := int(v) - int('a')
		freq[i] += 1
	}
	odd := 0
	for _, v := range freq {
		if 1 == v%2 {
			odd += 1
		}
	}
	// amount of centers 
	if odd > k {
		return false
	}
	return true
}

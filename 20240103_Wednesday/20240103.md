package main

func numberOfBeams(bank []string) int {
	// Two vars for prev and cur. row.
	// if curr is zero just skip it.
	// Number of beams between rows `prev*curr`;
	ans := 0
	prev, curr := 0, 0
	for _, row := range bank {
		// Calculate number of devices;
		curr = 0
		for _, val := range row {
			if '1' == val {
				curr += 1
			}
		}
		// Work only if curr != 0
		if 0 != curr {
			ans += (curr * prev)
			prev = curr
		}
	}
	return ans
}

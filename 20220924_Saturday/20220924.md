def path_sum(root, target_sum)
  @answer, @target = [], target_sum
  @path, @sum = [],0
  backtracking(root)
  @answer
end

def backtracking(node)
  return if node.nil?
  @path.push(node.val)
  @sum += node.val
  if @sum == @target && node.left.nil? && node.right.nil?
    @answer.push(@path.clone)
  end
  backtracking(node.left)
  backtracking(node.right)
  @path.pop
  @sum -= node.val
end

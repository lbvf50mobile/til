  def insert(node)
    if node.start >= @end1
      if @right.nil?
        @right = node
        return true
      end
      return @right.insert(node)
    elsif node.end1 <= @start
      if @left.nil?
        @left = node
        return true
      end
      return @left.insert(node)
    else
      return false
    end
  end

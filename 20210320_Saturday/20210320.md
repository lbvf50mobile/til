def remove_nth_from_end(head, n)
    list = []
    cur = head
    while cur
        list.push(cur)
        cur = cur.next
    end
    list.push(nil)
    size = list.size
    cur_index = size - 1 - n
    next_index = cur_index + 1
    prev_index = cur_index - 1
    return list[1] if 0 == cur_index
    list[prev_index].next = list[next_index]
    head
end

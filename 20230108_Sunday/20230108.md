def max_points(points)
  # Based on:
  # https://leetcode.com/problems/max-points-on-a-line/solution/
  n = points.size
  return 1 if 1 >= n
  ans = 2
  (0...n).each do |i|
    cnt = {}
    (0...n).each do |j|
      if i != j
        value = Math.atan2(points[j][1] - points[i][1], points[j][0] - points[i][0])
        cnt[value] ||= 0
        cnt[value] += 1
        ans = [ans,cnt.values.max + 1].max
      end
    end
  end
  return ans
end

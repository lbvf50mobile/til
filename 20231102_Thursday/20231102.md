def average_of_subtree(root)
  @ans = 0
  dfs(root)
  return @ans
end

# Return: [sum, total]
def dfs(x)
  return [0,0] if !x
  s1, t1 = dfs(x.left)
  s2, t2 = dfs(x.right)
  sum = s1+s2+x.val
  total = t1+t2+1
  av = sum/total
  @ans += 1 if av == x.val
  return [sum, total] 
end

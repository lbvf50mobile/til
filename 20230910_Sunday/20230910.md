def count_orders(n)
  # Based on:
  # https://leetcode.com/problems/count-all-valid-pickup-and-delivery-options/discuss/516941/Python-O(n)-with-detailed-explanations
  ans = 1
  mod = (10**9) + 7
  (1..n).each do |i|
    ans *= i * (2*i-1)
    ans %= mod
  end
  return ans
end

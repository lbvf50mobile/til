package main

func isReachableAtTime(sx int, sy int, fx int, fy int, t int) bool {
	// Based on:
	// https://leetcode.com/problems/determine-if-a-cell-is-reachable-at-a-given-time/solution/
	w := abs(sx - fx)
	h := abs(sy - fy)
	if 0 == w && 0 == h && 1 == t {
		return false
	}
	return t >= w && t >= h

}

func abs(a int) int {
	if a < 0 {
		return -a
	}
	return a
}

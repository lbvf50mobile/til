# @param {Integer} k

def find_kth_number(m, n, k)
  # Based on:
  # https://leetcode.com/problems/kth-smallest-number-in-multiplication-table/solution/
  @m,@n,@k = m,n,k
  lo, hi = 1, @m*@n
  while lo < hi
    mi = (lo+hi)/2
    if ! enough(mi)
      lo = mi + 1
    else
      hi = mi
    end
  end
  lo
end


def enough(x)
  count = 0
  (1..@m).each do |i|
    a = x/i
    a = @n if @n < a
    count += a
  end
  return count >= @k
end

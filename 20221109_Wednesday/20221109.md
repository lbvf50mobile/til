class StockSpanner
    def initialize()
      @sv = [10**7] # stack values.
      @si = [0] # stack indices.
      @i = 0
    end


=begin
    :type price: Integer
    :rtype: Integer
=end
    def next(price)
      @i += 1
      while (!@sv.empty?) && @sv.last <= price
        @sv.pop()
        @si.pop()
      end
      @sv.push(price)
      @si.push(@i)
      if 0 == @i
        return 1
      elsif 1 == @sv.size
        return @i
      else
        return @i - @si[-2]
      end
    end
end

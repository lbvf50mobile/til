# @param {Integer[]} nums
# @return {Integer}
def zero_filled_subarray(nums)
  zero_counter = 0
  answer = 0
  nums.each do |x|
    if 0 == x
      zero_counter += 1
      answer += zero_counter
    else
      zero_counter = 0
    end
  end
  return answer 
end

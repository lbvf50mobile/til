# @param {Character[]} letters
# @param {Character} target
# @return {Character}
def next_greatest_letter(letters, target)
  first = letters.first
  tmp  = letters.select{|x| x > target}
  if tmp.empty?
    first
  else
    tmp.first
  end
end

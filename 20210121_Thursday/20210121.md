import re
def solve(st,k): 
    count = 0
    while count < k and 0 < len(st):
        if re.search('a[^a]',st):
            st = re.sub('a([^a])',r'\1',st,1)
        else:
            for i in range(ord('a'),ord('z')+1):
                if chr(i) in st:
                    st = st.replace(chr(i),"",1)
                    break
        count += 1
    return st

```Ruby
# 507. Perfect Number
# https://leetcode.com/problems/perfect-number/
# @param {Integer} num
# @return {Boolean}
require "prime"
def check_perfect_number(num)
    sum = 0
    (1...num).each do |i|
        sum += i if 0 == num%i
    end
    sum == num
end
```

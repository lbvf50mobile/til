# @param {Integer[]} arr
# @param {Integer} m
# @param {Integer} k
# @return {Boolean}
def contains_pattern(arr, m, k)
    total_length = m*k
    last_starting_index = arr.size - total_length
    correct = ->x{
        (0...m).each do |i|
            el = arr[i+x]
            (1...k).each do |j|
                new_el = arr[i+x+j*m]
                return false if new_el != el
            end
        end
        true
    }
    (0..last_starting_index).each do |i|
        return true if correct[i]
    end
    false
end

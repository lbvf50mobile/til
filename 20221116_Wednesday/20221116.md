def guessNumber(n)
  # Errichto: Binary Search tutorial (C++ and Python)
  # https://youtu.be/GU7DpgHINWQ
  l,h = 1,n
  while l <= h
    m = l + (h-l)/2
    ans = guess(m)
    if 0 == ans
      return m
    elsif -1 == ans
      h = m - 1
    else
      l = m + 1
    end
  end
  raise "Cannot find an answer."
end

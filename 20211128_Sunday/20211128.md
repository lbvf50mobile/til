```Ruby
# @param {Integer[][]} graph
# @return {Integer[][]}
# [[4,3,1],[3,2,4,0],[3],[4],[]]
# The graph is not a DAG.
def all_paths_source_target(graph)
  @g = graph
  @visited, @ans = [], []
  bt(0)
  @ans
end

def bt(i)
  @visited.push(i)
  if i == @g.size - 1
    @ans.push(@visited.clone)
  else
    @g[i].each do |j|
      bt(j)
    end
  end
  @visited.pop()
end
```

  return if ! r

  arr.push(r.val) if (!r.left) && (!r.right)
  get_leafs(r.right,arr) if r.right
end

def leaf_similar(root1, root2)
  a,b = [],[]

  get_leafs(root2,b)
  a == b
end
def get_leafs(r,arr)
  stack = []
  while r || (!stack.empty?)
    if r
      stack.push(r)
      r = r.left
    else
      r = stack.pop
      if r.left.nil? && r.right.nil?
        arr.push(r.val)
      end
      r = r.right
    end
  end
end
